/* --------------------------------
   STEP 1 : CREATE DATABASE
--------------------------------*/
DROP DATABASE IF EXISTS company;
CREATE DATABASE company;
USE company;


/* --------------------------------
   STEP 2 : CREATE TABLE
--------------------------------*/
CREATE TABLE employee(
    eid INT,
    name VARCHAR(20),
    salary INT,
    department VARCHAR(20)
);

   STEP 3 : INSERT RECORDS
